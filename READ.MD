Big O Notations

// Time Complexity
// n linear
function summation(n) {
  let sum = 0;
  for (let i =1; i<=n; i++) {
    sum += i;
  }
  return sum;
}

//* Time Complexity
// Two loop -- Quadratic
// Three Loop -- Cubic
// logn - Logarithmic

//* Space Complexity
// O(1) - COnstant
// O(n) - Linear
// O(logn) - Logarithmic
